% Customizes the appearance of chapter titles using the KOMA-Script commands.
% This redefines the chapter format to include a styled box and line.
% Begins a section of code that changes internal LaTeX commands.
\makeatletter

% Redefines the chapter format.
\renewcommand*{\chapterformat}{
  % Makes sure \unitlength stays local
  \begingroup %
    % Sets the unit length for the picture environment to 1mm.
    \setlength{\unitlength}{1mm}%
    % Begins a picture environment with specified dimensions and offset.
    \begin{picture}(20,20)(0,3)
      % Sets the space between the frame and the chapter title to 0pt.
      \setlength{\fboxsep}{0pt}
      % Draws a white square of 20mm by 20mm at position (0,0).
      \put(0,0){\makebox(20,20){\rule{20\unitlength}{20\unitlength}}}%
      % Dividor line 
      \put(22,13){\linethickness{0.5pt}\line(1,0){\dimexpr
        \textwidth-24\unitlength\relax\@gobble}}
      % Chapter number    
      \put(-1.4,0.8){\makebox(18,18)[r]{%
          \color{white}\fontsize{22\unitlength}{28\unitlength}\selectfont\thechapter
          % Moves digit in the character cell to the right.
          %\kern-.05em 
        }}
      % Chapter title
      \put(20,13){\makebox(\dimexpr
          \textwidth-18\unitlength\relax\@gobble,\ht\strutbox\@gobble)[l]{%
            \ \normalsize\color{black}\chapapp~\thechapter
          }}
    % Space is needed here.
    \end{picture} 
  \endgroup
}

% Defines a new column type 'L' for left-aligned cells with fixed width.
% - \raggedright: Aligns text to the left without full justification.
% - \let\newline\\: Allows the use of \\ for line breaks within cells.
% - \arraybackslash: Restores the ability to use \\ at the end of cells.
% - m{#1}: Creates a fixed-width column with vertical centering.
% Source: http://tex.stackexchange.com/questions/12703/how-to-create-fixed-width-table-columns-with-text-raggedright-centered-raggedlef
\newcolumntype{L}[1]{>{\raggedright\let\newline\\\arraybackslash\hspace{0pt}}m{#1}}

% Defines a new column type 'C' for center-aligned cells with fixed width.
% Similar to 'L' but centers the text within the column.
\newcolumntype{C}[1]{>{\centering\let\newline\\\arraybackslash\hspace{0pt}}m{#1}}

% Defines a new column type 'R' for right-aligned cells with fixed width.
% Similar to 'L' but aligns the text to the right within the column.
\newcolumntype{R}[1]{>{\raggedleft\let\newline\\\arraybackslash\hspace{0pt}}m{#1}}

% 'quotemarks' command formats quotation marks in boldface.
% It defines a new command and environment for customized quotations within chapters.
% It takes one argument, which is the quotation mark symbol.
\newcommand{\quotemarks}[1]{{\bfseries{#1}}}

% 'chapter-quotation' environment creates a customized quotation block.
% It takes one argument, which can be used for additional formatting or notes.
\newenvironment{chapter-quotation}[1]
  {
    \begin{quotation}                     % Begins the quotation environment.
    \pushQED{#1}                          % Pushes the QED symbol (if any) for the end of the quotation.
    \noindent                             % Removes indentation at the beginning of the quotation.
    \slshape                              % Sets the font to slanted shape.
    \quotemarks{``}                       % Inserts the opening quotation marks in boldface.
    \hspace{-2pt}                         % Adjusts horizontal spacing for aesthetic alignment.
  }
  {
    \hspace{-2pt}                         % Adjusts horizontal spacing before the closing quotation marks.
    \quotemarks{''}                       % Inserts the closing quotation marks in boldface.
    \fontsize{10}{10}\normalfont\\{ }\\   % Sets the font size and resets to normal font, adding line breaks.
    \hspace*{\fill}---\popQED             % Adds a right-aligned em-dash and pops the QED symbol (if any).
    \end{quotation}                       % Ends the quotation environment.
  }
